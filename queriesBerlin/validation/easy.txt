WHERE {
	?offer bsbm:product <http://www4.wiwiss.fu-berlin.de/bizer/bsbm/v01/instances/dataFromProducer9/Product416> .
	?offer bsbm:vendor ?vendor .
    ?offer dc:publisher ?vendor .

	FILTER (?deliveryDays <= 3)
	?offer bsbm:price ?price .

}
